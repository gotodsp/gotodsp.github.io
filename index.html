def g(x):
    return x % 100000000 + x // 100000000

def f(x):
    if x == 1:
        return 3
    elif x % 2 == 0:
        return g(f(x / 2) * f(x / 2))
    elif x % 2 == 1:
        return g(f(1) * f(x - 1))

if __name__ == '__main__':
    print(f(50))
